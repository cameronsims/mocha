Mocha - console.h
LOG
#include "console.h"
// console log example 1
template<typename T>
void cout(T subject) {
    mocha::log << subject;
}
// console log example 2
int main() {
    mocha::log << "This is a float: " << 85.3f << '\n';
}
mocha::log is a singleton which functions like std::cout from the standard library
CIN
#include "console.h"
// console in example
int main() {
    int l;
    mocha::cin >> l;
    mocha::log << l << '\n';
}
mocha::cin is a singleton which functions like std::cin from the standard library
